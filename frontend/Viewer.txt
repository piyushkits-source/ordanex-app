🚀 1. DocumentViewer.tsx (MAIN ENGINE)

import PdfRenderer from "./PdfRenderer";
import ImageRenderer from "./ImageRenderer";
import SpreadsheetRenderer from "./SpreadsheetRenderer";
import EdiRenderer from "./EdiRenderer";
import WordRenderer from "./WordRenderer";

export default function DocumentViewer({
  file,
  boxes,
  selectedField,
  onSelectField,
}: any) {

  if (!file) return <div className="empty-box">No file available</div>;

  const type = resolveViewerType(file.original_file_name);

  switch (type) {
    case "pdf":
      return <PdfRenderer {...{ file, boxes, selectedField, onSelectField }} />;

    case "image":
      return <ImageRenderer {...{ file, boxes, selectedField, onSelectField }} />;

    case "spreadsheet":
      return <SpreadsheetRenderer file={file} />;

    case "edi":
      return <EdiRenderer file={file} />;

    case "word":
      return <WordRenderer file={file} />;

    default:
      return <div className="empty-box">Unsupported format</div>;
  }
}

function resolveViewerType(name: string) {
  const ext = name?.toLowerCase().split(".").pop();

  if (["pdf"].includes(ext)) return "pdf";
  if (["png", "jpg", "jpeg", "tiff"].includes(ext)) return "image";
  if (["xls", "xlsx", "csv"].includes(ext)) return "spreadsheet";
  if (["doc", "docx"].includes(ext)) return "word";
  if (["edi", "txt"].includes(ext)) return "edi";

  return "unknown";
}

import { Document, Page } from "react-pdf";
import { useState } from "react";

export default function PdfRenderer({
  file,
  boxes = [],
  selectedField,
  onSelectField,
}: any) {

  const [numPages, setNumPages] = useState(1);

  const fileUrl = `/api/files/${file.file_id}/download`;

  return (
    <div className="pdf-viewer-shell">

      <Document file={fileUrl} onLoadSuccess={(pdf) => setNumPages(pdf.numPages)}>
        {Array.from(new Array(numPages), (_, i) => (
          <div key={i} className="pdf-page">
            <Page pageNumber={i + 1} width={800} />

            {/* Bounding Boxes */}
            <div className="bbox-overlay">
              {boxes
                .filter((b: any) => b.page === i + 1)
                .map((box: any, idx: number) => (
                  <div
                    key={idx}
                    className={`bbox ${selectedField === box.field_name ? "selected" : ""}`}
                    style={{
                      left: box.x,
                      top: box.y,
                      width: box.width,
                      height: box.height,
                    }}
                    onClick={() => onSelectField(box.field_name)}
                  />
                ))}
            </div>

          </div>
        ))}
      </Document>
    </div>
  );
}

📊 4. SpreadsheetRenderer.tsx

import { useEffect, useState } from "react";

export default function SpreadsheetRenderer({ file }: any) {
  const [data, setData] = useState<any[]>([]);

  useEffect(() => {
    fetch(`/api/files/${file.file_id}/preview`)
      .then((r) => r.json())
      .then(setData);
  }, []);

  return (
    <div className="table-wrap">
      <table className="items-table">
        <tbody>
          {data.map((row, i) => (
            <tr key={i}>
              {Object.values(row).map((cell: any, j) => (
                <td key={j}>{cell}</td>
              ))}
            </tr>
          ))}
        </tbody>
      </table>
    </div>
  );
}




📄 5. EdiRenderer.tsx

import { useEffect, useState } from "react";

export default function EdiRenderer({ file }: any) {
  const [text, setText] = useState("");

  useEffect(() => {
    fetch(`/api/files/${file.file_id}/download`)
      .then((r) => r.text())
      .then(setText);
  }, []);

  return (
    <div className="edi-view">
      <pre>{text}</pre>
    </div>
  );
}
